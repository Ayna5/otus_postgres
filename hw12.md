## Работа с join'ами, статистикой

```sql
CREATE SCHEMA IF NOT EXISTS otus;

CREATE TABLE IF NOT EXISTS otus.user (
    user_id    SERIAL PRIMARY KEY,
    first_name TEXT,
    last_name  TEXT
);

INSERT INTO otus.user(first_name, last_name)
VALUES ('Иван', 'Иванов'),
       ('Владислав', 'Александров'),
       ('Михаил', 'Огурцов');

CREATE TABLE IF NOT EXISTS otus.phone (
    id     SERIAL PRIMARY KEY,
    number TEXT
);

INSERT INTO otus.phone(number)
VALUES ('89277234567'),
       ('89996764545'),
       ('89277234568');

CREATE TABLE IF NOT EXISTS otus.address_book (
    address_book_id SERIAL PRIMARY KEY,
    user_first_name TEXT,
    user_last_name  TEXT,
    phone_number BIGINT
    FOREIGN KEY (user_first_name) REFERENCES user (first_name)
    FOREIGN KEY (user_last_name) REFERENCES user (last_name)
    FOREIGN KEY (phone_number) REFERENCES phone (number)        
);

INSERT INTO otus.address_book(user_first_name, user_last_name, phone_number)
VALUES ('Иван', 'Иванов', '89277234567'),
       ('Владислав', 'Александров', '89996764545'),
       ('Михаил', 'Огурцов', '89277234568'); 
```
